docker内拨号

（头条是主机拨号）

如果之前有服务正在运行则停掉容器
docker kill `docker ps -a --format "table {{.ID}}\t{{.Names}}" |grep -E 'ksp2p_|pcdn_|cp2p_'|awk '{print $1}'`

拨号镜像地址

registry.cn-hangzhou.aliyuncs.com/pcdn-images/pppoe:v5.9

拨号脚本

curl -sSL https://automateplus.oss-cn-hangzhou.aliyuncs.com/docker_pppoe_multieth.sh|bash -s – -c 拨号镜像版本

创建拨号虚拟网卡脚本

curl -sSL https://automateplus.oss-cn-hangzhou.aliyuncs.com/host_pppoe_vlan.sh|sh

可以使用以下脚本直接直接执行操作 \

查看网卡，有state up的，并且有ip的网卡是主机网卡（内网ip），有state up，并且没有ip的是拨号网卡

拨号步骤

  1. 创建拨号文件夹
    mkdir /opt/pppoe
  2. 创建拨号配置文件
    /opt/pppoe/pppoe_user.txt
    文件格式：
    vlanid 拨号账户 账户密码 拨号网卡名 带宽大小（M）
    例：
    novlan 77561213412 00000000 ens3 100
    novlan 77539581907 00000000 ens3 100
    novlan 77515478098 00000000 ens3 100
    novlan 77531248304 00000000 ens3 100
    注意：第一列为vlanid，如果线路未设置vlan，则值为“novlan”
  3. 检验拨号是否成功

hyp -l $uuid -p “/opt/test/test.sh”

mkdir /opt/pppoe
cd /opt/pppoe
vi /opt/pppoe/pppoe_user.txt

function del_pppoe()
{
docker rm -f docker ps -a --format "table {{.ID}}\t{{.Names}}" |grep pppoe_|awk '{print $1}'
#清空 echo “”

cat /var/spool/cron/root
echo > /var/spool/cron/root

#删除
cat /etc/rc.local
sed -i '/host_pppoe_vlan.sh/d' /etc/rc.local



rm -rf  /etc/sysconfig/network-scripts/ifcfg-macv*
for j in $(/usr/sbin/ip addr  | grep -Eo "\w+\.[0-9]+@" | awk -F'@' '{print $1}')
do
    #echo $j
    #ifdown ${j}
    /usr/sbin/ip link delete ${j}
done
for i in $(/usr/sbin/ip addr | grep -Eo "macv[0-9]+")
do
    #echo $i
    #ifdown-macvlan $i
    /usr/sbin/ip link delete ${i}
done

}
del_pppoe

function create_pppoe()
{
declare -A param_map
param_map=([p2p]=”-v /etc/localtime:/etc/localtime” [pcdn]=”–cpus=1 -m 1G –memory-swap -1 -v /etc/localtime:/etc/localtime” [cp2p]=”-v /etc/localtime:/etc/localtime” [vdp]=”–cpus=1 -v /etc/localtime:/etc/localtime -v /opt/oort:/opt/oort” [ksp2p]=”-v /opt/hyops:/opt/hyops -v /etc/salt:/etc/salt” [wsp2p]=”-v /etc/localtime:/etc/localtime” [pop2p]=”-v /etc/localtime:/etc/localtime -v /etc/salt/minion_id:/etc/salt/minion_id”)
ver=v5.9
docker kill docker ps -a --format "table {{.ID}}\t{{.Names}}" |grep -E 'ksp2p_|pcdn_|cp2p_'|awk '{print $1}'
if [[ $(docker ps | grep -c “pppoe_”) -eq 0 ]];then
curl -sSL https://automateplus.oss-cn-hangzhou.aliyuncs.com/docker_pppoe_multieth.sh|bash -s – -c ${ver}
fi

###################重启服务###########
mountstr="-v /etc/resolv.conf:/etc/resolv.conf"
envstr=""
for str_service in $(docker ps -a | awk '{print $NF}' | awk -F'_' '{print $1}'| uniq | grep -vE 'dcache|NAMES|monitor|ipes-manager|pppoe|k8s');do
    docker_version=$(docker ps  -a | grep "${str_service}_" | head -n1 | awk   '{print $2}' |  awk  -F':' '{print $2}')
    service_Num=$(docker ps -a | grep -c ${str_service}_)
    param=${param_map["$str_service"]}
    pppoe_Num=$(docker ps | grep -c "pppoe_")
    num_a=1
    num_b=1
    while [[ $num_a -le $service_Num ]]; do
        if [[ $num_b -eq $((${pppoe_Num}+1)) ]]; then
            num_b=1
        else
            path=`docker inspect ${str_service}_${num_a} |grep Mounts -A 30|grep "Source"|grep "/data[1-9]*[0-9]*"|sed -r 's/".*?": "(.*?)",/\1/'|sed 's/ //g'`
            echo ${str_service}_${num_a} pppoe_$num_b
            if [[ $str_service = "cp2p" ]]; then
                docker rm -f ${str_service}_${num_a}
                docker run --restart=always --net=container:pppoe_$num_b --name=${str_service}_${num_a}  $param $envstr  $mountstr -v ${path}:/storage -d registry.cn-hangzhou.aliyuncs.com/pcdn-images/coldp2p:${docker_version}
            elif [[ $str_service = "pcdn" ]]; then
                docker rm -f ${str_service}_${num_a}
                echo "docker run --restart=always --net=container:pppoe_$num_b --name=${str_service}_${num_a}   $param $envstr  $mountstr -v $path:/storage -d registry.cn-hangzhou.aliyuncs.com/pcdn-images/pcdn:${docker_version}"
                docker run --restart=always --net=container:pppoe_$num_b --name=${str_service}_${num_a}   $param $envstr  $mountstr -v $path:/storage -d registry.cn-hangzhou.aliyuncs.com/pcdn-images/pcdn:${docker_version}             
            elif [[ $str_service = "ksp2p" ]];then
                docker rm -f ${str_service}_${num_a}
                start_port=$((17999+num_a))
                res=`netstat -lnp|grep "$start_port" |wc -l`
                while [[ $res -gt 0 ]];do
                    strat_port=$((strat_port+1))
                    res=`netstat -lnp|grep $strat_port |wc -l`
                done
                envstr=" -e ConName=${str_service}_${num_a} -e BasePort=$start_port "
                docker run --restart=always --net=container:pppoe_$num_b --name=${str_service}_${num_a}  -v /opt/hyops:/opt/hyops -v /etc/salt:/etc/salt  $envstr ${mountstr} -v ${path}:/storage -d registry.cn-hangzhou.aliyuncs.com/p2pcdn/ksp2p:${docker_version}
            elif [[ $str_service == "vdp" ]]; then
                envstr=" -e OORT_ENV_POD_NAME=btvdp-${num_a} "
                mountstr="-v /opt/hyops/dns/vdp/resolv.conf:/etc/resolv.conf"
                docker rm -f ${str_service}_${num_a}
                echo "docker run --restart=always --net=container:pppoe_$num_b --name=${str_service}_${num_a} $param $envstr $mountstr -v $path:/storage -d registry.cn-hangzhou.aliyuncs.com/p2pcdn/btvdp:${docker_version}"
                docker run --restart=always --net=container:pppoe_$num_b --name=${str_service}_${num_a} $param $envstr $mountstr -v $path:/storage -d registry.cn-hangzhou.aliyuncs.com/p2pcdn/btvdp:${docker_version}
            fi
            ((num_a++))
            ((num_b++))
        fi
    done
done

docker start monitor

}
